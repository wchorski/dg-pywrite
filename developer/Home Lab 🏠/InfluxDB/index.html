Install

Install InfluxDB 2.7 | Docker

version: '3.3'
services:
  influxdb:
    image: influxdb:2.7
    container_name: influxdb
    ports:
      - '8086:8086'
    volumes:
      - ./db:/var/lib/influxdb2
      - ./config.yml:/etc/influxdb2/config.yml

create blank file config.yml in the root of directory. Look here for config options

docker compose up -d

visit http://localhost:8060 to set up user, bucket, and key

yarn add @influxdata/influxdb-client @influxdata/influxdb-client-apis
