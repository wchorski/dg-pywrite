  1. grab zsh if you want all the pretty icons
sudo apt-get update
sudo apt-get install zsh
zsh
# press (2)

# make zsh default shell
chsh -s $(which zsh)
chsh -s /bin/bash
  1. installer script
curl -sS https://starship.rs/install.sh | sh
  1. Add the following to the end of ~/.zsh
eval "$(starship init zsh)"
  1. add config file
mkdir -p ~/.config && touch ~/.config/starship.toml
